def esc(x): return x.replace('\\','\\\\').replace('"','\\"').replace('\n','\\n')

def node_end(s, open_idx):           # quote/brace-aware scan from the node's '{'
    depth=0; i=open_idx; instr=None
    while i<len(s):
        c=s[i]
        if instr:
            if c=='\\': i+=2; continue
            if c==instr: instr=None
        elif c in "'\"": instr=c
        elif c=='{': depth+=1
        elif c=='}':
            depth-=1
            if depth==0: return i
        i+=1
    return -1

ins=upd=miss=0
for nid,sql in SQL.items():
    pat="{id:'%s'"%nid
    idx=s.find(pat)
    if idx<0: miss+=1; print('NOT FOUND:',nid); continue
    end=node_end(s,idx)
    node=s[idx:end+1]
    esql=esc(sql)
    m=re.search(r'sql:"(?:[^"\\]|\\.)*"', node)
    if m:
        newnode=node[:m.start()]+'sql:"%s"'%esql+node[m.end():]; upd+=1
    else:
        newnode=node.replace(pat, pat+',sql:"%s"'%esql, 1); ins+=1
    s=s[:idx]+newnode+s[end+1:]

open(HTML,'w',encoding='utf-8').write(s)
print('nodes inserted:',ins,'| updated:',upd,'| missing:',miss)
