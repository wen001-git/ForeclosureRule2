    cur=info['current']

    if 'datediff' in t or '天数' in t or '+ datediff' in t: return '天数重算'
    if 'datadic' in t or 'decode' in t or '解码' in t: return 'decode解码'
    if 'case when' in t or 'concat' in t or ' case ' in t or '取反' in t: return 'CASE派生'
    if 'null' in t and ('常量' in t or 'constant' in t or 'hardcod' in t): return '常量NULL'
    if 'rename' in t or '改名' in t: return '改名'
    if 'passthrough' in t or '直传' in t or 'direct copy' in t or 'direct' in t or 'upsert pass' in t or 'pass-through' in t: return '直传'

# ---- 主链 hops → 分层列 ----
def src_of(hops, prefixes):
    out = []
    for h in hops:
        t = str(h.get('t','')); c = str(h.get('c','')); rule = h.get('rule',''); code = h.get('code','')
        if any(t.startswith(p) or t == p for p in prefixes):
            seg = (c if c else '') + (' ('+rule+')' if rule else '')
            if code: seg += ' ['+code+']'
            out.append(seg.strip())
    return ' ; '.join([x for x in out if x])

def eg_worked(meta, field):
    TEMP='tempfc.temp_basic_data_fcl'; FCL='port.basic_data_loan_fcl'
    FORE='port.basic_data_loan_foreclosure'; SYNC='bpms.sync_loan_foreclosure'
    VIEW='bpms.biz_data_view_loan_details_foreclosure'; NZ='7727003984'; CA='7727000806'
    def v(mt,fd,ln): return str(LE_VALS.get(mt,{}).get(ln,{}).get(fd,'∅NULL'))
    if field=='servicer':
        return "例：按 UNION 分支赋常量——Newrez 分支→'Newrez'、Carrington 分支→'Carrington'、Capecodfive 分支→'Capecodfive'"
    if field=='daysinfc' and meta in (TEMP,FCL):
        return ("例 Newrez 7727003984：直传源 daysinfc=%s（servicer 自报，不重算）　vs　"
                "Carr 7727000806：计算 datediff(referral %s → 快照 %s)+1=%s（仅 FCL 活跃，否则 NULL）"
                % (v(meta,'daysinfc',NZ), v(meta,'referral_start_date',CA), v(meta,'dataasof',CA), v(meta,'daysinfc',CA)))
    if field=='activefcflag' and meta in (TEMP,FCL):
        return ("例 Newrez 7727003984：cast(activefcflag as int)=%s　vs　"
                "Carr 7727000806：CASE fcl_flag='Active'→1 否则 NULL（本笔=%s）；CC5 同理用 foreclosure_flag"
                % (v(meta,'activefcflag',NZ), v(meta,'activefcflag',CA)))
    if field=='fcstage' and meta in (TEMP,FCL):
        return ("例 Newrez 7727003984：直传 fcstage=%s　vs　Carr 7727000806：取 fcl_sub_status=%s（Carr 无独立 fcstage 列）"
                % (v(meta,'fcstage',NZ), v(meta,'fcstage',CA)))

                % v(meta,'summary_foreclosure_status',NZ))

def _passthrough(meta, field, loan):    # 直传/改名 → 引用上游该笔格（无 =）
    imm=IMM_UP.get(meta)
    if imm and field in REG.get(imm,{}).get('field2row',{}): return cref(imm,field,loan)
    if meta==TEMP and field in TEMP_RM and TEMP_RM[field] in REG.get(SRC,{}).get('field2row',{}): return cref(SRC,TEMP_RM[field],loan)
    u=UPM.get((meta,field))
    if u and u[0] in REG and _PASS.search(u[2] or '') and u[1] in REG[u[0]]['field2row']: return cref(u[0],u[1],loan)
    if meta==FCL and field in REG.get(SRC,{}).get('field2row',{}): return cref(SRC,field,loan)
    return None
def f_status(loan):                     # CASE 止赎状态（输入 ⑩；null-guard → Carr 空时 ∅NULL）
    af=cref(FCL,'activefcflag',loan); de=cref(FCL,'fcremovaldesc',loan)
    if not af or not de: return None
    case='IF(IFERROR(VALUE(%s),-1)=1,"Active Foreclosure",IF(AND(IFERROR(VALUE(%s),-1)=0,NOT(OR(%s="∅NULL",%s="∅空串",%s=""))),"Closed Foreclosure:"&%s,""))'%(af,af,de,de,de,de)
    return 'IF(%s="∅NULL","∅NULL",%s)'%(af,case)
def f_type(meta):
    def g(loan):
        j=cref(meta,'summary_judicial_foreclosure',loan)
        return 'IF(%s="∅NULL","∅NULL",IF(IFERROR(VALUE(%s),-1)>=1,"Judicial","Non Judicial"))'%(j,j) if j else None
    return g
def f_days(loan):                       # ⑱ days 实时重算 = ⑪存储 + (TODAY-⑪dataasof)
    d=cref(FORE,'summary_days_in_fcl',loan); a=cref(FORE,'dataasof',loan)
    if not d or not a: return None
    return 'IF(%s="∅NULL","∅NULL",IFERROR(VALUE(%s),0)+(TODAY()-DATEVALUE(%s)))'%(a,d,a)
def f_actual(loan, field):              # ⑳ actual_X = TO_DAYS(timeline_X)-TO_DAYS(nextduedate)
    core=field[len('actual_'):-len('_days')]; tl='timeline_%s_date'%core
    if tl in _FULLHIST: return None     # set_date 类格含「值\n多天提示」两行，DATEVALUE 无法解析 → 跳过(出注)
    if tl not in REG.get(VIEW,{}).get('field2row',{}): return None
    t=cref(VIEW,tl,loan); nd=cref(VIEW,'nextduedate',loan)
    if not t or not nd: return None
    return 'IF(OR(%s="∅NULL",%s="∅NULL"),"∅NULL",DATEVALUE(%s)-DATEVALUE(%s))'%(t,nd,t,nd)
def f_days_carr(meta, loan):            # Carr daysinfc = datediff(referral,snap)+1（同表）
    rf=cref(meta,'referral_start_date',loan); da=cref(meta,'dataasof',loan)
    if not rf or not da: return None
    return 'IF(OR(%s="∅NULL",%s="∅NULL"),"∅NULL",DATEVALUE(%s)-DATEVALUE(%s)+1)'%(da,rf,da,rf)

 (SYNC,'summary_days_in_fcl'):f_days}
def formula_for(meta, field, loan):     # -> (cell_text, is_formula)
    svc=LOAN_SVC[loan]
    if field=='servicer': return ('="%s"'%svc, True)
    if field=='daysinfc' and svc=='Carrington' and meta in (TEMP,FCL):
        b=f_days_carr(meta,loan); return ('='+b, True) if b else (NOTE_GEN, False)
    if meta==TEMP and svc=='Carrington' and field in CARR_DERIVE:   # ⑨ Carr 派生字段：引用表底「Carr 源输入」块 → 活公式
        b=carr_formula(field)
        if b: return (b, True)
    if (meta,field) in COMPUTED:
        b=COMPUTED[(meta,field)](loan); return ('='+b, True) if b else (NOTE_GEN, False)
    if meta==VIEW and field.startswith('actual_') and field.endswith('_days'):
        b=f_actual(loan,field); return ('='+b, True) if b else (NOTE_GEN, False)
    if field in _FULLHIST: return ('首见 min(dataasof)：活公式见本页底部「首见 min 演示」块 / D 页', False)
    if meta==TEMP and svc=='Carrington':   # 剩余 Carr 字段=该 UNION 分支恒 NULL 常量（无对应 raw 源列）
        return ('Carr 分支=NULL 常量（CREATE_BASIC_FCL Carrington 分支无此源列，pool:1574-1611）', False)
    a=_passthrough(meta,field,loan)
    if a: return ('='+a, True)
    return (NOTE_GEN, False)
